Reo.js <api: template()>

演示：

HTML:

name: {{personal.unit}}
address: {{personal.address}}
QQ: {{email.seconds.num}}{{email.seconds.signals}}

JS:

reo.template({
  target : '#container',
   template : '#temp',
   val : {
      personal : {
        unit : "The China",
        address : "Shanghai Baoshanqu Qihua rd",
      }
   }
 });
